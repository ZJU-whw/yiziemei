CREATE OR REPLACE PROCEDURE TL_TSSH.PRO_FXGL_FXSPCK_DZXX
/*
 * 风险商品出口电子信息提取
 * 20250331，根据省局李裕军局长意见，对市场采购业务不做风险商品出口短信提醒
 */
AS
  LD_BEGINDATE      DATE;
BEGIN
  SELECT MAX(T.SJTB_SJ)-3 INTO LD_BEGINDATE FROM CKTS_WBSJ_HG_BGD T;

  FOR CUR_FXGZ IN (SELECT GZ_MC,GZ_FXMS,GZ_FW_SWJG,GZ_SPDM,GZ_SPMC,GZ_GGXH,GZ_CKKA,GZ_CKGB,GZ_HYD,GZ_DJQ,GZ_DJZ,GZ_ZJQ,GZ_ZJZ,GZ_YXQZ,GZ_DXTXBZ
                     FROM FXGL_SZ_SPFXCK
                    WHERE GZ_FW_DATA IN ('1','3') --规则适用业务范围：1出口电子信息+3同时适用
                      AND QYBJ='Y' --规则启用标记：Y
                      AND NVL(GZ_YXQZ,DATE'2100-12-31')>SYSDATE --规则有效期止：NULL或大于当前日期
  ) LOOP
    BEGIN
      --规则中所有设置为空，无效规则挑过
      IF CUR_FXGZ.GZ_SPDM IS NULL AND CUR_FXGZ.GZ_SPMC IS NULL AND CUR_FXGZ.GZ_GGXH IS NULL
         AND CUR_FXGZ.GZ_CKKA IS NULL AND CUR_FXGZ.GZ_CKGB IS NULL AND CUR_FXGZ.GZ_HYD IS NULL
         AND CUR_FXGZ.GZ_DJQ IS NULL AND CUR_FXGZ.GZ_DJZ IS NULL AND CUR_FXGZ.GZ_ZJQ IS NULL AND CUR_FXGZ.GZ_ZJZ IS NULL THEN
        CONTINUE;
      END IF;

      -- 根据风险规则扫描出口电子信息
      FOR CUR_FXSJ IN (SELECT T.DJXH,T.CKBGDH,T.MYLAJ,T.CKNY,T.CKSP_DM,T.CKRQ_1,SUBSTR(T.CKBGDH,1,18) AS CKBGD18,T.HGSPMC,
                              S.NSRDZDAH,S.SWJG_DM,NVL(S.SHXYNO,S.NSRDJNO) AS NSRSBH,S.NSRMC
                         FROM TL_TSSH.CKTS_WBSJ_HG_BGD T
                        INNER JOIN TL_BJTS.GS_DJ_CKTMSDAB S ON S.CPCODE=T.DJXH
                        WHERE T.JGFS_DM<>'1039'
                          AND T.SJTB_SJ>=LD_BEGINDATE
                          AND (T.TSSWJG_DM_1 LIKE CUR_FXGZ.GZ_FW_SWJG||'%')
                          AND (CUR_FXGZ.GZ_SPDM IS NULL OR
                              (SUBSTR(CUR_FXGZ.GZ_SPDM,1,1)='!' AND NOT REGEXP_LIKE(T.CKSP_DM,SUBSTR(CUR_FXGZ.GZ_SPDM,2))) OR
                              (SUBSTR(CUR_FXGZ.GZ_SPDM,1,1)<>'!' AND REGEXP_LIKE(T.CKSP_DM,CUR_FXGZ.GZ_SPDM)))
                          AND (CUR_FXGZ.GZ_SPMC IS NULL OR REGEXP_LIKE(T.HGSPMC,CUR_FXGZ.GZ_SPMC))
                          AND (CUR_FXGZ.GZ_GGXH IS NULL OR REGEXP_LIKE(T.GGXH,CUR_FXGZ.GZ_GGXH))
                          AND (CUR_FXGZ.GZ_CKKA IS NULL OR REGEXP_LIKE(T.HGGQKA_DM,CUR_FXGZ.GZ_CKKA))
                          AND (CUR_FXGZ.GZ_CKGB IS NULL OR REGEXP_LIKE(T.MYGDQSZ_DM,CUR_FXGZ.GZ_CKGB))
                          AND (CUR_FXGZ.GZ_HYD IS NULL OR
                              (SUBSTR(CUR_FXGZ.GZ_HYD,1,1)='!' AND NOT REGEXP_LIKE(T.HZDWDQ_DM,SUBSTR(CUR_FXGZ.GZ_HYD,2))) OR
                              (SUBSTR(CUR_FXGZ.GZ_HYD,1,1)<>'!' AND REGEXP_LIKE(T.HZDWDQ_DM,CUR_FXGZ.GZ_HYD)))
                          AND (CUR_FXGZ.GZ_DJQ IS NULL OR DECODE(T.CKSL,0,0,T.MYLAJ/T.CKSL)>CUR_FXGZ.GZ_DJQ)
                          AND (CUR_FXGZ.GZ_DJZ IS NULL OR DECODE(T.CKSL,0,0,T.MYLAJ/T.CKSL)<CUR_FXGZ.GZ_DJZ)
                          AND (CUR_FXGZ.GZ_ZJQ IS NULL OR T.MYLAJ>CUR_FXGZ.GZ_ZJQ)
                          AND (CUR_FXGZ.GZ_ZJZ IS NULL OR T.MYLAJ<CUR_FXGZ.GZ_ZJZ)
      ) LOOP
        BEGIN
          -- TL_ADMIN.YJ_DATA_YJXX，态势感知风险管理，所有风险数据往这张表写数据
          INSERT INTO TL_ADMIN.YJ_DATA_YJXX(
                      TBPC,ID,SBID,NSRDZDAH,CJ_DATE,YJCODE,ZBCODE,YJ_OBJECT,YJ_COUNT,YJ_AMT,YJ_RECORD,YJ_MSG,SBYM,SWFLAG,QYFLAG,BMDFLAG)
               SELECT 0,SEQ_YJ_DATA_YJXX.NEXTVAL,0,CUR_FXSJ.NSRDZDAH,SYSDATE,'140','14001',CUR_FXSJ.CKSP_DM,1,CUR_FXSJ.MYLAJ,CUR_FXSJ.CKBGDH,
                      '风险商品出口(规则：'||NVL(CUR_FXGZ.GZ_MC,CUR_FXGZ.GZ_FXMS)||')，报关单号('||CUR_FXSJ.CKBGDH||
                      ')，商品代码('||CUR_FXSJ.CKSP_DM||')，商品名称('||CUR_FXSJ.CKSP_DM||')，出口美元金额'||CUR_FXSJ.MYLAJ||')',
                      CUR_FXSJ.CKNY,1,0,0
                 FROM DUAL
                WHERE NOT EXISTS (SELECT 1
                                    FROM TL_ADMIN.YJ_DATA_YJXX A
                                   WHERE A.SBID=0 AND A.NSRDZDAH=CUR_FXSJ.NSRDZDAH
                                     AND A.YJCODE='140' AND A.ZBCODE='14001'
                                     AND A.YJ_RECORD=CUR_FXSJ.CKBGDH)
                  AND NOT EXISTS (SELECT 1
                                      FROM TL_ADMIN.YJ_CS_BMD B
                                      LEFT JOIN TL_ADMIN.YJ_CS_BMD_SUB C ON C.BMDID=B.ID
                                     WHERE B.NSRDZDAH=CUR_FXSJ.NSRDZDAH
                                       AND B.YJCODE='140'
                                       AND (B.OBJFLAG='0' OR C.YJ_OBJECT=CUR_FXGZ.GZ_MC));
          -- TL_TSSH.JKGL_GY_FXSM_HGCK，原三三智检风险出口数据，待迁移
          INSERT INTO JKGL_GY_FXSM_HGCK(DJXH,CKBGDH,SMRQ,FXGZ)
               SELECT CUR_FXSJ.DJXH,CUR_FXSJ.CKBGDH,SYSDATE,CUR_FXGZ.GZ_MC
                 FROM DUAL
                WHERE NOT EXISTS (SELECT 1
                                    FROM JKGL_GY_FXSM_HGCK A
                                   WHERE A.DJXH=CUR_FXSJ.DJXH
                                     AND A.CKBGDH=CUR_FXSJ.CKBGDH)
                  AND NOT EXISTS (SELECT 1
                                      FROM TL_ADMIN.YJ_CS_BMD B
                                      LEFT JOIN TL_ADMIN.YJ_CS_BMD_SUB C ON C.BMDID=B.ID
                                     WHERE B.NSRDZDAH=CUR_FXSJ.NSRDZDAH
                                       AND B.YJCODE='140'
                                       AND (B.OBJFLAG='0' OR C.YJ_OBJECT=CUR_FXGZ.GZ_MC));
          -- TL_TSSH.MSG_PUSH_DATA，风险出口数据的短信提醒
          -- 从原PRO_JKGL_FXCKSP存储过程迁移到这来，原biztype为“风险出口”，现改为“风险出口电子信息”
          IF CUR_FXGZ.GZ_DXTXBZ='Y' THEN
            -- 20240819，插入风险出口的短信提醒信息，
            -- 风险出口短信提醒按18位报关单号汇总
            INSERT INTO MSG_PUSH_DATA(ID,SWJG_DM,NSRSBH,NSRMC,BIZTYPE,BIZKEY,QDSJ,JZSJ,YWBZ,SJTBSJ,IDLY)
                 SELECT SEQ_MSG_PUSH_DATA.NEXTVAL,CUR_FXSJ.SWJG_DM,CUR_FXSJ.NSRSBH,CUR_FXSJ.NSRMC,'风险出口电子信息',
                        CUR_FXSJ.CKBGD18,CUR_FXSJ.CKRQ_1,NULL,CUR_FXGZ.GZ_MC,SYSDATE,'1'
                   FROM DUAL
                  WHERE NOT EXISTS (SELECT 1
                                      FROM MSG_PUSH_DATA A
                                     WHERE A.BIZTYPE='风险出口电子信息'
                                       AND A.BIZKEY=CUR_FXSJ.CKBGD18)
                    AND NOT EXISTS (SELECT 1
                                      FROM TL_ADMIN.YJ_CS_BMD B
                                      LEFT JOIN TL_ADMIN.YJ_CS_BMD_SUB C ON C.BMDID=B.ID
                                     WHERE B.NSRDZDAH=CUR_FXSJ.NSRDZDAH
                                       AND B.YJCODE='140'
                                       AND (B.OBJFLAG='0' OR C.YJ_OBJECT=CUR_FXGZ.GZ_MC));
          END IF;
          COMMIT;
        END;
      END LOOP;
    END;
  END LOOP;

  MERGE INTO FXGL_SZ_SPFXJH A
       USING (SELECT T.FHXXBUUID,T.FUHQFRQ,T.GHQYNSRSBH_1,T.GHFQYMC_1,'一年以上未回函' AS GZ_MC,
                     T.FAHDSWJGMC||TO_CHAR(T.FAHQFRQ,'yyyy-mm-dd')||'发函，至今尚无明确回函意见' AS GZ_FXMS
                FROM TL_ADMIN.YJ_CS_YCHD_JS T
               WHERE T.FUHLX_DM IS NULL AND SYSDATE-T.FAHQFRQ>365) B
          ON (B.FHXXBUUID=A.FHXXBUUID AND B.GZ_MC=A.GZ_MC)
        WHEN NOT MATCHED THEN
          INSERT (ID,TSSWJG_DM,TSSWJG_MC,TBR,TBRQ,GZ_FW_SWJG,GZ_FW_DATA,GZ_MC,GZ_FXMS,GZ_GHQYSH,GZ_GHQYMC,GZ_DXTXBZ,QYBJ,FHXXBUUID)
          VALUES (SEQ_FXGL_SZ_SPFXJH.NEXTVAL,'13300000000','浙江省税务局','SYSTEM',SYSDATE,'133','2',B.GZ_MC,B.GZ_FXMS,B.GHQYNSRSBH_1,B.GHFQYMC_1,'N','Y',B.FHXXBUUID);
  COMMIT;
  MERGE INTO FXGL_SZ_SPFXJH A
       USING (SELECT T.FHXXBUUID,T.FUHQFRQ,T.GHQYNSRSBH_1,T.GHFQYMC_1,'回函存在异常业务' AS GZ_MC,
                     T.FAHDSWJGMC||TO_CHAR(T.FAHQFRQ,'yyyy-mm-dd')||'发函，对方税务机关回函存在异常业务' AS GZ_FXMS
                FROM TL_ADMIN.YJ_CS_YCHD_JS T
               WHERE T.FUHLX_DM='2') B
          ON (B.FHXXBUUID=A.FHXXBUUID AND B.GZ_MC=A.GZ_MC)
        WHEN NOT MATCHED THEN
          INSERT (ID,TSSWJG_DM,TSSWJG_MC,TBR,TBRQ,GZ_FW_SWJG,GZ_FW_DATA,GZ_MC,GZ_FXMS,GZ_GHQYSH,GZ_GHQYMC,GZ_DXTXBZ,QYBJ,FHXXBUUID,GZ_YXQZ)
          VALUES (SEQ_FXGL_SZ_SPFXJH.NEXTVAL,'13300000000','浙江省税务局','SYSTEM',SYSDATE,'133','2',B.GZ_MC,B.GZ_FXMS,B.GHQYNSRSBH_1,B.GHFQYMC_1,'N','Y',B.FHXXBUUID,ADD_MONTHS(B.FUHQFRQ,12));
  COMMIT;
  /*MERGE INTO FXGL_SZ_SPFXJH A
       USING (SELECT T.FHXXBUUID,T.FUHQFRQ,T.GHQYNSRSBH_1,T.GHFQYMC_1,'回函正常其他情况不予退税' AS GZ_MC,
                     T.FAHDSWJGMC||TO_CHAR(T.FAHQFRQ,'yyyy-mm-dd')||'发函，对方税务机关回函正常，经核查其他情况不予退税' AS GZ_FXMS
                FROM TL_ADMIN.YJ_CS_YCHD_JS T
               WHERE T.FUHLX_DM<>'2' AND T.FUHCLYJ_DM='3') B
          ON (B.FHXXBUUID=A.FHXXBUUID AND B.GZ_MC=A.GZ_MC)
        WHEN NOT MATCHED THEN
          INSERT (ID,TSSWJG_DM,TSSWJG_MC,TBR,TBRQ,GZ_FW_SWJG,GZ_FW_DATA,GZ_MC,GZ_FXMS,GZ_GHQYSH,GZ_GHQYMC,GZ_DXTXBZ,QYBJ,FHXXBUUID,GZ_YXQZ)
          VALUES (SEQ_FXGL_SZ_SPFXJH.NEXTVAL,'13300000000','浙江省税务局','SYSTEM',SYSDATE,'133','2',B.GZ_MC,B.GZ_FXMS,B.GHQYNSRSBH_1,B.GHFQYMC_1,'N','Y',B.FHXXBUUID,ADD_MONTHS(B.FUHQFRQ,12));
  COMMIT;*/
  UPDATE FXGL_SZ_SPFXJH A
     SET A.GZ_YXQZ=(SELECT FUHQFRQ FROM TL_ADMIN.YJ_CS_YCHD_JS B WHERE B.FHXXBUUID=A.FHXXBUUID)
   WHERE A.FHXXBUUID IS NOT NULL AND A.GZ_MC='一年以上未回函';
  COMMIT;

  RETURN;
END;
