-- MySQL 5.5 版本：计算一组数据的标准值（剔除坏点后取平均）
-- 对应 Oracle 11g 脚本：方法函数-1-Oracle.txt（tl_tssh.F_MY_STDAVG_TMPTB）
--
-- 说明：
-- 1. 原对象是 Oracle 函数，但其核心是动态执行调用方传入的取数 SQL（p_datasql）。
--    MySQL 5.5 的存储函数内禁止使用动态 SQL（PREPARE/EXECUTE），因此改写为
--    存储过程，计算结果通过 OUT 参数 p_stdavg 返回（对应原函数的 return）。
--    调用方式：
--        CALL f_my_stdavg_tmptb('select zb_total as obj from jkgl_pz_jkm', @avg);
--        SELECT @avg;
-- 2. 输入取数脚本的要求与原脚本一致：数据列必须别名为 obj。示例（已按要求
--    去掉 schema 前缀）：
--        select zb_total as obj from jkgl_pz_jkm
--        select zb_val+0 as obj from jkgl_data_zb_jgb where zb_val is not null and zb_val<>0
--        （Oracle 的 to_number(zb_val) 在 MySQL 5.5 中可用 zb_val+0 或
--          CAST(zb_val AS DECIMAL(18,2)) 代替）
-- 3. p_datasql 与原脚本相同是原样拼接执行的 SQL，注入风险同原脚本，由调用方
--    保证内容可信。注意 MySQL 的派生表必须有别名，动态 SQL 中已补别名 dt，
--    传入的查询末尾不要带分号。
-- 4. Oracle 的 GLOBAL TEMPORARY TABLE 是库级共享对象，原脚本先查 user_tables
--    判断存在再 drop、用完再 drop，并发调用会互相冲突；MySQL 的 TEMPORARY TABLE
--    按连接隔离，这里用 DROP TEMPORARY TABLE IF EXISTS + CREATE TEMPORARY TABLE
--    一步到位，天然并发安全，ON COMMIT DELETE ROWS 无需对应（过程末尾直接删表）。
-- 5. Oracle 的 STDDEV 是“样本标准差”（分母 n-1），对应 MySQL 的 STDDEV_SAMP；
--    MySQL 的 STDDEV/STD 是总体标准差（分母 n），不能直接照搬函数名。
--    另：只剩 1 行数据时 Oracle STDDEV 返回 0、MySQL STDDEV_SAMP 返回 NULL，
--    两者经 nvl/IFNULL(x,0)=0 判断后同样退出循环，最终结果一致。
-- 6. SQL%ROWCOUNT → ROW_COUNT()，须在 EXECUTE / UPDATE 之后立即读取。
--    UPDATE 的 ROW_COUNT() 统计“实际变更”的行数，本场景 WHERE 已限定 ybbz='1'
--    （更新后必然变化），与 Oracle 统计“匹配”行数的行为一致。
-- 7. number(18,2) → DECIMAL(18,2)，赋值时同样按 2 位小数四舍五入，迭代行为一致；
--    循环上限（v_deeps>3，至多 4 轮）、剔除阈值（偏差/标准差 > 2）均按原样保留。
-- 8. 原脚本注释写“偏差在一倍以上剔除”，代码实际用 > 2（两倍标准差），
--    按“保留原处理逻辑”的要求以代码为准，注释原样保留。
-- 9. 建议在默认字符集为 utf8 的库中创建，执行前先 USE 到目标库。

SET NAMES utf8;

-- USE 目标数据库;   -- 例如：USE tl_tssh;

DELIMITER $$

DROP PROCEDURE IF EXISTS f_my_stdavg_tmptb$$
CREATE PROCEDURE f_my_stdavg_tmptb(
    IN  p_datasql VARCHAR(2000),   -- 数据组查询脚本，数据列必须别名为 obj
    OUT p_stdavg  DECIMAL(18, 2)   -- 标准值（对应原函数 return number）
)
-- 计算一组数据的标准值
-- 区别于算数平均值，是一组数中剔除和标准差偏离较大的坏点后，再取平均值
BEGIN
    DECLARE v_stddev DECIMAL(18, 2);
    DECLARE v_avg    DECIMAL(18, 2);
    DECLARE v_deeps  INT;   -- 嵌套计算深度
    DECLARE v_id     INT;
    DECLARE v_inscnt INT;   -- 装载行数（对应插入后的 SQL%ROWCOUNT）

    -- 创建临时表（连接级私有，替代原 GLOBAL TEMPORARY TABLE + user_tables 存在性检查）
    DROP TEMPORARY TABLE IF EXISTS tmp_stddev;
    CREATE TEMPORARY TABLE tmp_stddev (
        id   INT,
        val  DECIMAL(18, 2),
        ybbz CHAR(1)
    );

    SET v_id = 1;
    SET v_deeps = 0;

    -- 首先，将数组写入临时表（p_datasql 为运行期内容，须用 PREPARE 动态执行；
    -- MySQL 派生表必须有别名，故补别名 dt）
    SET @tmp_sql_stdavg = CONCAT(
        'insert into tmp_stddev(id,val,ybbz) ',
        'select ', v_id, ', obj, ''1'' from (', p_datasql, ') dt');
    PREPARE stmt_stdavg FROM @tmp_sql_stdavg;
    EXECUTE stmt_stdavg;
    SET v_inscnt = ROW_COUNT();
    DEALLOCATE PREPARE stmt_stdavg;

    IF v_inscnt = 0 THEN
        SET v_avg = NULL;
    ELSE
        calc_loop: LOOP
            IF v_deeps > 3 THEN
                LEAVE calc_loop;
            END IF;
            SET v_deeps = v_deeps + 1;

            -- 计算标准差（Oracle STDDEV=样本标准差，对应 STDDEV_SAMP）
            SELECT STDDEV_SAMP(val)
              INTO v_stddev
              FROM tmp_stddev
             WHERE id = v_id
               AND ybbz = '1';

            -- 计算平均值
            SELECT AVG(val)
              INTO v_avg
              FROM tmp_stddev
             WHERE id = v_id
               AND ybbz = '1';

            IF IFNULL(v_stddev, 0) = 0 OR IFNULL(v_avg, 0) = 0 THEN
                LEAVE calc_loop;
            END IF;

            -- 和平均值的偏差 和标准差相比，偏差在一倍以上的剔除掉。
            UPDATE tmp_stddev t
               SET t.ybbz = '0'
             WHERE t.id = v_id
               AND ABS(t.val - v_avg) / v_stddev > 2
               AND t.ybbz = '1';

            IF ROW_COUNT() = 0 THEN
                LEAVE calc_loop;
            END IF;
        END LOOP calc_loop;

        SELECT AVG(val)
          INTO v_avg
          FROM tmp_stddev
         WHERE id = v_id
           AND ybbz = '1';
    END IF;

    DROP TEMPORARY TABLE tmp_stddev;

    SET p_stdavg = v_avg;
END$$

DELIMITER ;
